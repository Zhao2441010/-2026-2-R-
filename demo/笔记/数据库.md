@Embeddable//定义嵌入主键
public class Adopt implements java.io.Serializable {
    @Column(name = "cat_id")
    private Long catId;

    @Column(name = "user_id")
    private Long userId;

    public Adopt() {
    }

    public Adopt(Long catId, Long userId) {
        this.catId = catId;
        this.userId = userId;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o) return true;
        if (o == null || getClass() != o.getClass()) return false;
        Adopt adopt = (Adopt) o;
        return catId.equals(adopt.catId) && userId.equals(adopt.userId);
    }

    @Override
    public int hashCode() {
        return java.util.Objects.hash(catId, userId);
    }

    public Long getCatId() {
        return catId;
    }

    public void setCatId(Long catId) {
        this.catId = catId;
    }

}

@Entity
@Table(name = "adopt")
public class Adopt {
    
    @EmbeddedId
    private UCid uc;

    @ManyToOne(fetch = FetchType.LAZY)
    //lazy加载，只有在真正需要使用user对象时才会去数据库中查询对应的user数据，避免了不必要的数据库查询，提高了性能
    @JoinColumn(name = "user_id", insertable = false, updatable = false)
    private User user;

    @ManyToOne(fetch = FetchType.LAZY)
    @JoinColumn(name = "cat_id", insertable = false, updatable = false)
    private Cat cat;
//这里的private不会在数据库中添加user除了id的信息

}


写映射关系@XXX时,可以方便获得外键对象信息,但是必须创建完整对象
如果不需要,不加外键映射即可

@ManyToMany,多对多关系
如果关联表还有附加信息,不能用manytomany,拆成两个onetomany
    @ManyToMany(fetch = FetchType.LAZY)

        inverseJoinColumns = @JoinColumn(name = "course_id") // 对方在中间表的外键
    )
    private Set<Course> courses = new HashSet<>();
